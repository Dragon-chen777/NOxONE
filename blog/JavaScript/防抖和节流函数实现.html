function throttle(fn, hold = 500) {
	let timer = null
	let start = +new Date()
	return function (...args) {
		timer && clearTimeout(timer)
		let end = +new Date()
		if (end - start >= hold) {
			start = end
			fn.apply(this, args)
		} else timer = setTimeout(() => fn.apply(this, args), hold)
	}
}
